-- 注册码表（id,注册码，disable‘true’|‘false’）
DROP TABLE IF EXISTS `register_code`;
CREATE TABLE register_code(
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `code` VARCHAR(5) NOT NULL UNIQUE COMMENT '注册码',
  `disable` ENUM('true','false') NOT NULL DEFAULT 'false' COMMENT '是否不可用'
);
